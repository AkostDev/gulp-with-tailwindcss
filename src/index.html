Commands

Use the following commands to get started.

// To Install Dev Depedencies
npm install

// To generate tailwind config file with tailwind utilites styles
npm run init

// To start development and server for live preview
npm run dev

// To generate minifed files for live server
npm run build

Options

To change the path of files and destination/build folder or font weights check package.json file

"options": {
  "config": {
    "tailwindjs": "./tailwind.js"
  },
  "paths": {
    "root": "./",
    "src": {
      "base": "./src",
      "css": "./src/css",
      "js": "./src/js",
      "sass": "./src/sass",
      "img": "./src/img"
    },
    "dist": {
      "base": "./dist",
      "css": "./dist/css",
      "js": "./dist/js",
      "img": "./dist/img"
    },
    "build": {
      "base": "./build"
    }
  },
  "fontweights": "400,500,600,700,900"
}
